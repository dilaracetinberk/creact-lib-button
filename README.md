## Install

```bash
npm install --save creact-lib-button
```

## Usage

```jsx
import React, { Component } from 'react'

import MyComponent from 'creact-lib-button'
import 'creact-lib-button/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
